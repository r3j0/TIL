```py
# 1759 : 암호 만들기
import sys
input = sys.stdin.readline

l, c = map(int, input().rstrip().split())
chars = sorted(list(input().rstrip().split()))

def backtracking(now, start, cnt):
    if cnt == l:
        now_str = ''
        m = 0
        done1 = 0
        done2 = 0
        while (1 << m) <= now:
            if (1 << m) & now: 
                now_str += chars[m]

                if chars[m] in ['a', 'e', 'i', 'o', 'u']: done1 += 1
                else: done2 += 1
            
            m += 1

        if done1 >= 1 and done2 >= 2: print(now_str)

        return

    for i in range(start, c):
        backtracking(now | (1 << i), i + 1, cnt + 1)

backtracking(0, 0, 0)
```
