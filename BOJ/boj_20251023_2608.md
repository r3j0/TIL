# 2608번: 로마 숫자
문자열 구현으로 풀이했다. 로마자의 규칙을 잘 이해해서 구현하는게 포인트이다.

```py
# 2608: 로마 숫자
import sys
input = sys.stdin.readline

rome = {'M':1000, 'CM':900, 'D':500, 'CD':400, 'C':100, 'XC':90, 'L':50, 'XL':40, 'X':10, 'IX':9, 'V':5, 'IV':4, 'I':1}
def RomeToDeci(string):
    result = 0
    i = 0
    while i < len(string):
        # IV, IX, XL, XC, CD, CM
        if i < len(string) - 1 and rome.get(string[i] + string[i+1]):
            result += rome[string[i] + string[i+1]]
            i += 1
        else:
            result += rome[string[i]]
        i += 1
    
    return result

def DeciToRome(num):
    result = ""
    for k, v in list(rome.items()):
        if num - v >= 0:
            result += k * (num // v)
            num %= v
    return result


a = input().rstrip()
b = input().rstrip()

res = RomeToDeci(a) + RomeToDeci(b)
print(res, DeciToRome(res), sep='\n')
```
