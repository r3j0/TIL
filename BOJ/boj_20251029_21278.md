# 21278번: 호석이 두 마리 치킨
플로이드-워셜 알고리즘을 이용하여 풀이했다.

```py
# 21278 : 호석이 두 마리 치킨
import sys
input = sys.stdin.readline

n, m = map(int, input().rstrip().split())

dp = [[float('inf') for _ in range(n+1)] for _ in range(n+1)]
for i in range(n+1): dp[i][i] = 0
for _ in range(m):
    u, v = map(int, input().rstrip().split())
    dp[u][v] = 1
    dp[v][u] = 1

for k in range(1, n+1):
    for a in range(1, n+1):
        for b in range(1, n+1):
            dp[a][b] = min(dp[a][b], dp[a][k] + dp[k][b])

min_dist = float('inf')
min_dist_city = [-1, -1]

for a in range(1, n):
    for b in range(a+1, n+1):
        now_dist = sum([min(dp[k][a]+dp[a][k], dp[k][b]+dp[b][k]) for k in range(1, n+1)])
        if min_dist > now_dist:
            min_dist = now_dist
            min_dist_city = [a, b]

print(' '.join(map(str, min_dist_city + [min_dist])))
```
