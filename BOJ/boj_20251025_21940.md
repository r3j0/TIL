# 21940번: 가운데에서 만나기
플로이드-워셜 알고리즘을 이용하여, 친구들의 도시와 특정 도시의 최대 왕복 시간이 최소인 도시를 찾는다.

```py
# 21940 : 가운데에서 만나기
import sys
input = sys.stdin.readline

n, m = map(int, input().rstrip().split())

dp = [[float('inf') for _ in range(n+1)] for _ in range(n+1)]
for i in range(1, n+1): dp[i][i] = 0

for _ in range(m):
    a, b, t = map(int, input().rstrip().split())
    dp[a][b] = t

for k in range(1, n+1):
    for a in range(1, n+1):
        for b in range(1, n+1):
            dp[a][b] = min(dp[a][b], dp[a][k] + dp[k][b])

k = int(input().rstrip())
arr = list(map(int, input().rstrip().split()))

min_dist = float('inf')
min_dist_city = []
for i in range(1, n+1):
    now_max_dist = 0
    for a in arr: now_max_dist = max(now_max_dist, dp[a][i] + dp[i][a])

    if now_max_dist <= min_dist:
        if now_max_dist == min_dist:
            min_dist_city.append(i)
        else:
            min_dist_city = [i]
        
        min_dist = now_max_dist
        
print(' '.join(map(str, min_dist_city)))
```
