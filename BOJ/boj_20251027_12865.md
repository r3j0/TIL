# 12865번: 평범한 배낭
Knapsack DP 를 사용하여 풀이했다.

```py
# 12865 : 평범한 배낭
import sys
input = sys.stdin.readline

n, k = map(int, input().rstrip().split())
arr = [list(map(int, input().rstrip().split())) for _ in range(n)]

dp = [[0 for _ in range(n+1)] for _ in range(k+1)]

for weight in range(1, k+1):
    for thing in range(1, n+1):
        now_weight = arr[thing-1][0]
        now_value = arr[thing-1][1]

        dp[weight][thing] = dp[weight][thing-1]
        if now_weight <= weight:
            dp[weight][thing] = max(dp[weight][thing], dp[weight-now_weight][thing-1] + now_value)

print(dp[k][n])
```
