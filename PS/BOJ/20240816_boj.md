## [14698 : 전생했더니 슬라임 연구자였던 건에 대하여 (Hard) ( GOLD 4 )](https://www.acmicpc.net/problem/14698)
```py
# 14698 : 전생했더니 슬라임 연구자였던 건에 대하여 (Hard)
import sys
import heapq
input = sys.stdin.readline

t = int(input().rstrip())
for _ in range(t):
  n = int(input().rstrip())
  arr = list(map(int, input().rstrip().split()))
  pq = []
  for i in range(n): heapq.heappush(pq, arr[i])

  ans = 1
  plus = 0
  while len(pq) != 1:
    first = heapq.heappop(pq)
    second = heapq.heappop(pq)
    if (first * second) > 0: plus = 1
    ans = (ans * (first * second)) % 1000000007
    heapq.heappush(pq, first * second)

  print(ans if plus == 1 else 1)
```

우선순위 큐를 이용해 작은 수부터 곱했다.

## [1141 : 접두사 ( SILVER 1 )](https://www.acmicpc.net/problem/1141)
```py
# 1141 : 접두사
import sys
input = sys.stdin.readline

n = int(input().rstrip())
arr = [input().rstrip() for _ in range(n)]
arr.sort(key=lambda x:(-len(x)))
cnt = 1
for i in range(1, n):
  for j in range(i):
    if arr[j][:len(arr[i])] == arr[i]:
      break
    if j == i - 1:
      cnt += 1

print(cnt)
```

문자열이 긴 순서대로 정렬하여, 처음부터 차례대로 이전 문자열의 접두사가 될 수 있는지를 체크한다.

# 19949 : 영재의 시험
import sys
input = sys.stdin.readline
sys.setrecursionlimit(500000)

arr = list(map(int, input().rstrip().split()))

def rec(now):
  if len(now) == len(arr):
    ans = 0
    for i in range(len(arr)):
      if now[i] == arr[i]: ans += 1
    if ans >= 5: return 1
    else: return 0

  cnt = 0
  for i in range(1, 5+1):
    if len(now) < 2 or (len(now) >= 2 and (not (now[-2] == i and now[-1] == i))):
      now.append(i)
      cnt += rec(now)
      now.pop()
  return cnt

print(rec([]))
```

# 3425 : 고스택
import sys
from collections import deque
input = sys.stdin.readline
stack = deque()
def num(x): stack.append(x)
def pop(): 
  if len(stack) == 0: return 0
  stack.pop()
  return 1
def inv(): 
  if len(stack) == 0: return 0
  stack[-1] *= -1
  return 1
def dup(): 
  if len(stack) == 0: return 0
  stack.append(stack[-1])
  return 1
def swp(): 
  if len(stack) < 2: return 0
  first = stack.pop()
  second = stack.pop()
  stack.append(first)
  stack.append(second)
  return 1
def add():
  if len(stack) < 2: return 0
  first = stack.pop()
  second = stack.pop()
  if abs(first + second) > 10**9: return 0
  stack.append(first + second)
  return 1
def sub():
  if len(stack) < 2: return 0
  first = stack.pop()
  second = stack.pop()
  if abs(second - first) > 10**9: return 0
  stack.append(second - first)
  return 1
def mul():
  if len(stack) < 2: return 0
  first = stack.pop()
  second = stack.pop()
  if abs(first * second) > 10**9: return 0
  stack.append(first * second)
  return 1
def div():
  if len(stack) < 2: return 0
  first = stack.pop()
  second = stack.pop()
  if first == 0: return 0
  res = 0
  if (second > 0 and first < 0):
    res = -(second // (-first))
  elif (second < 0 and first > 0):
    res = -((-second) // first)
  else:
    res = second // first
  if abs(res) > 10**9: return 0
  stack.append(res)
  return 1
def mod():
  if len(stack) < 2: return 0
  first = stack.pop()
  second = stack.pop()
  if first == 0: return 0
  res = 0
  if (second > 0 and first < 0):
    res = (second % (-first))
  elif (second < 0 and first > 0):
    res = -((-second) % first)
  elif second < 0:
    res = -(-(second) % -(first))
  else:
    res = second % first
  if abs(res) > 10**9: return 0
  stack.append(res)
  return 1

tc = 0
while True:
  order = []
  r = 0
  while True:
    now = input().rstrip()
    while now == '': now = input().rstrip()
    if now == 'END': break
    if now == 'QUIT':
      r = 1
      break
    order.append(now)
  if r == 1: break
  tc += 1
  
  n = int(input().rstrip())
  for _ in range(n):
    now = int(input().rstrip())
    stack = deque()
    stack.append(now)

    e = 0
    for o in order:
      if o[:3] == 'NUM': 
        num(int(o[4:]))
      elif o == 'POP':
        if not pop():
          print('ERROR')
          e = 1
          break
      elif o == 'INV':
        if not inv():
          print('ERROR')
          e = 1
          break
      elif o == 'DUP':
        if not dup():
          print('ERROR')
          e = 1
          break
      elif o == 'SWP':
        if not swp():
          print('ERROR')
          e = 1
          break
      elif o == 'ADD':
        if not add():
          print('ERROR')
          e = 1
          break
      elif o == 'SUB':
        if not sub():
          print('ERROR')
          e = 1
          break
      elif o == 'MUL':
        if not mul():
          print('ERROR')
          e = 1
          break
      elif o == 'DIV':
        if not div():
          print('ERROR')
          e = 1
          break
      elif o == 'MOD':
        if not mod():
          print('ERROR')
          e = 1
          break
    if e == 0:
      if len(stack) != 1: print('ERROR')
      else: print(stack[0])

  print()
```
