## 1. Codeforces Round 944 (Div. 4) [D. Binary Cut](https://codeforces.com/problemset/problem/1971/D)
```py
# D. Binary Cut
import sys
input = sys.stdin.readline

t = int(input().rstrip())
for _ in range(t):
  string = input().rstrip()
  group = []
  now_group = [string[0], 1]
  for i in range(1, len(string)):
    if now_group[0] == string[i]:
      now_group[1] += 1
    else:
      group.append(now_group)
      now_group = [string[i], 1]
  group.append(now_group)
  res = len(group)
  for i in range(len(group)-1):
    if group[i][0] == '0' and group[i+1][0] == '1':
      res -= 1
      break
  print(res)
```
그룹 구현으로 풀이 했다. sorted binary string 을 만들기 위해선 0 그룹 -> 1 그룹 쌍을 한 개의 쌍으로 만들어야 한다.

## 2. Codeforces Round 941 (Div. 2) [B. Rectangle Filling]()
```py
# B. Rectangle Filling
import sys
input = sys.stdin.readline

t = int(input().rstrip())
for _ in range(t):
  n, m = map(int, input().rstrip().split())
  maps = [list(input().rstrip()) for _ in range(n)]
  perfect_mode = 0
  perfect_fill_i = -1
  if n == m == 1: print("YES")
  else:
    for i in range(n):
      for j in range(m):
        if i == 0 and j == 0: continue
        if maps[i][j] == maps[0][0]:
          if perfect_mode == 1:
            perfect_fill_i = i
            break
          else:
            if j == m - 1:
              perfect_mode = 1
              perfect_fill_i = i
              break
    if perfect_fill_i == n - 1: print('YES')
    else:
      perfect_fill_i = -1
      perfect_mode = 0
      for i in range(n):
        for j in range(m-1, -1, -1):
          if i == 0 and j == m-1: continue
          if maps[i][j] == maps[0][m-1]:
            if perfect_mode == 1:
              perfect_fill_i = i
              break
            else:
              if j == 0:
                perfect_mode = 1
                perfect_fill_i = i
                break
      if perfect_fill_i == n - 1: print('YES')
      else: 
        perfect_fill_i = n
        perfect_mode = 0
        for i in range(n - 1, -1, -1):
          for j in range(m):
            if i == n - 1 and j == 0: continue
            if maps[i][j] == maps[n - 1][0]:
              if perfect_mode == 1:
                perfect_fill_i = i
                break
              else:
                if j == m - 1:
                  perfect_mode = 1
                  perfect_fill_i = i
                  break
        if perfect_fill_i == 0: print('YES')
        else:
          perfect_fill_i = n
          perfect_mode = 0
          for i in range(n - 1, -1, -1):
            for j in range(m - 1, -1, -1):
              if i == n - 1 and j == m - 1: continue
              if maps[i][j] == maps[n - 1][m - 1]:
                if perfect_mode == 1:
                  perfect_fill_i = i
                  break
                else:
                  if j == 0:
                    perfect_mode = 1
                    perfect_fill_i = i
                    break
          if perfect_fill_i == 0: print('YES')
          else: print('NO')
```
