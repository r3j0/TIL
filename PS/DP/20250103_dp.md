# 습격자 초라기 (Dynamic Programming)
Date : 2025. 01. 03

[백준 1006 : 습격자 초라기 (PLATINUM 3)](https://www.acmicpc.net/problem/1006) 문제를 풀이하였습니다.  

## 사고 과정
- 한 특수 소대는 특정 한 구역과 동시에 인접한 다른 한 구역을 커버할 수 있는 건, 2xn 구역을 1x1, 2x1, 1x2 로 타일링하는 것과 유사하다! 동시에 두 구역을 커버할 땐 합이 w보다 작거나 같은지 확인한다.
- dp는 한 열을 기준으로 배열을 만들되, 5가지 형태의 경우의 수를 구분한다.

## 풀이 코드
```py
# 1006 : 습격자 초라기
import sys
input = sys.stdin.readline

t = int(input().rstrip())
for _ in range(t):
    n, w = map(int, input().rstrip().split())
    arr = [list(map(int, input().rstrip().split())) for _ in range(2)]
    ans = float('inf')

    # 1. 시작을 연결하지 않고 그냥 둔다.
    dp = [[float('inf') for _ in range(5)] for _ in range(n)]
    dp[0][0] = 2
    
    if n >= 2:
        # 0
        for j in range(5):
            dp[1][0] = min(dp[1][0], dp[0][j] + 2)

        # 1
        if arr[0][1] + arr[1][1] <= w:
            for j in range(5):
                dp[1][1] = min(dp[1][1], dp[0][j] + 1)

    for i in range(2, n):
        # 0
        for j in range(5):
            dp[i][0] = min(dp[i][0], dp[i-1][j] + 2)
        # 1
        if arr[0][i] + arr[1][i] <= w:
            for j in range(5):
                dp[i][1] = min(dp[i][1], dp[i-1][j] + 1)
        
        # 2
        if arr[0][i-1] + arr[0][i] <= w:
            for j in [0, 3]:
                dp[i][2] = min(dp[i][2], dp[i-1][j] + 1)
        
        # 3
        if arr[1][i-1] + arr[1][i] <= w:
            for j in [0, 2]:
                dp[i][3] = min(dp[i][3], dp[i-1][j] + 1)
        
        # 4
        if arr[0][i-1] + arr[0][i] <= w and arr[1][i-1] + arr[1][i] <= w:
            dp[i][4] = min(dp[i][4], dp[i-1][0])

    ans = min(ans, min(dp[n-1]))
    if n >= 2:
        if arr[0][n-1] + arr[0][0] <= w:
            for j in [0, 3]:
                ans = min(ans, dp[n-1][j] - 1)
        if arr[1][n-1] + arr[1][0] <= w:
            for j in [0, 2]:
                ans = min(ans, dp[n-1][j] - 1)
        if arr[0][n-1] + arr[0][0] <= w and arr[1][n-1] + arr[1][0] <= w:
            ans = min(ans, dp[n-1][0] - 2)
    
    # 2. 시작을 아래 위로 연결한다. 
    dp = [[float('inf') for _ in range(5)] for _ in range(n)]
    if arr[0][0] + arr[1][0] <= w: dp[0][1] = 1
    
    if n >= 2:
        # 0
        for j in range(5):
            dp[1][0] = min(dp[1][0], dp[0][j] + 2)
            
        # 1
        if arr[0][1] + arr[1][1] <= w:
            for j in range(5):
                dp[1][1] = min(dp[1][1], dp[0][j] + 1)

    for i in range(2, n):
        # 0
        for j in range(5):
            dp[i][0] = min(dp[i][0], dp[i-1][j] + 2)
        # 1
        if arr[0][i] + arr[1][i] <= w:
            for j in range(5):
                dp[i][1] = min(dp[i][1], dp[i-1][j] + 1)
        
        # 2
        if arr[0][i-1] + arr[0][i] <= w:
            for j in [0, 3]:
                dp[i][2] = min(dp[i][2], dp[i-1][j] + 1)
        
        # 3
        if arr[1][i-1] + arr[1][i] <= w:
            for j in [0, 2]:
                dp[i][3] = min(dp[i][3], dp[i-1][j] + 1)
        
        # 4
        if arr[0][i-1] + arr[0][i] <= w and arr[1][i-1] + arr[1][i] <= w:
            dp[i][4] = min(dp[i][4], dp[i-1][0])
    
    ans = min(ans, min(dp[n-1]))

    if n >= 2:
        # 3. 시작을 아래 위 둘 다 각각 연결한다.
        dp = [[float('inf') for _ in range(5)] for _ in range(n)]
        if arr[0][0] + arr[0][1] <= w and arr[1][0] + arr[1][1] <= w: 
            dp[1][4] = 2

            for i in range(2, n):
                # 0
                for j in range(5):
                    dp[i][0] = min(dp[i][0], dp[i-1][j] + 2)
                # 1
                if arr[0][i] + arr[1][i] <= w:
                    for j in range(5):
                        dp[i][1] = min(dp[i][1], dp[i-1][j] + 1)
                
                # 2
                if arr[0][i-1] + arr[0][i] <= w:
                    for j in [0, 3]:
                        dp[i][2] = min(dp[i][2], dp[i-1][j] + 1)
                
                # 3
                if arr[1][i-1] + arr[1][i] <= w:
                    for j in [0, 2]:
                        dp[i][3] = min(dp[i][3], dp[i-1][j] + 1)
                
                # 4
                if arr[0][i-1] + arr[0][i] <= w and arr[1][i-1] + arr[1][i] <= w:
                    dp[i][4] = min(dp[i][4], dp[i-1][0])
            
            ans = min(ans, min(dp[n-1]))

        # 4. 시작을 위만 연결한다.
        dp = [[float('inf') for _ in range(5)] for _ in range(n)]
        if arr[0][0] + arr[0][1] <= w: 
            dp[1][2] = 3

            for i in range(2, n):
                # 0
                for j in range(5):
                    dp[i][0] = min(dp[i][0], dp[i-1][j] + 2)
                # 1
                if arr[0][i] + arr[1][i] <= w:
                    for j in range(5):
                        dp[i][1] = min(dp[i][1], dp[i-1][j] + 1)
                
                # 2
                if arr[0][i-1] + arr[0][i] <= w:
                    for j in [0, 3]:
                        dp[i][2] = min(dp[i][2], dp[i-1][j] + 1)
                
                # 3
                if arr[1][i-1] + arr[1][i] <= w:
                    for j in [0, 2]:
                        dp[i][3] = min(dp[i][3], dp[i-1][j] + 1)
                
                # 4
                if arr[0][i-1] + arr[0][i] <= w and arr[1][i-1] + arr[1][i] <= w:
                    dp[i][4] = min(dp[i][4], dp[i-1][0])
            
            ans = min(ans, min(dp[n-1]))
            if arr[1][n-1] + arr[1][0] <= w:
                for j in [0, 2]:
                    ans = min(ans, dp[n-1][j] - 1)

        # 5. 시작을 아래만 연결한다.
        dp = [[float('inf') for _ in range(5)] for _ in range(n)]
        if arr[1][0] + arr[1][1] <= w: 
            dp[1][3] = 3

            for i in range(2, n):
                # 0
                for j in range(5):
                    dp[i][0] = min(dp[i][0], dp[i-1][j] + 2)
                # 1
                if arr[0][i] + arr[1][i] <= w:
                    for j in range(5):
                        dp[i][1] = min(dp[i][1], dp[i-1][j] + 1)
                
                # 2
                if arr[0][i-1] + arr[0][i] <= w:
                    for j in [0, 3]:
                        dp[i][2] = min(dp[i][2], dp[i-1][j] + 1)
                
                # 3
                if arr[1][i-1] + arr[1][i] <= w:
                    for j in [0, 2]:
                        dp[i][3] = min(dp[i][3], dp[i-1][j] + 1)
                
                # 4
                if arr[0][i-1] + arr[0][i] <= w and arr[1][i-1] + arr[1][i] <= w:
                    dp[i][4] = min(dp[i][4], dp[i-1][0])
            
            ans = min(ans, min(dp[n-1]))
            if arr[0][n-1] + arr[0][0] <= w:
                for j in [0, 3]:
                    ans = min(ans, dp[n-1][j] - 1)
    
    print(ans)
```
